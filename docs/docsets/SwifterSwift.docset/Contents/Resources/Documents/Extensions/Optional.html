  • run(_:)

    SwifterSwift: Runs a block to Wrapped if not nil

    let foo: String? = nil
foo.run { unwrappedFoo in
    // block will never run sice foo is nill
    print(unwrappedFoo)
}

let bar: String? = "bar"
bar.run { unwrappedBar in
    // block will run sice bar is not nill
    print(unwrappedBar) -> "bar"
}
